1. Ուղղանկյան բարձրությունը 23սմ է, երկարությունը 46սմ: Հաշվել ուղղանկյան մակերեսը:

Ըստ s = a * b բանաձևի

let s, a, b;

a = 23;

b = 46;

s = a * b;

console.log(s);

2. Հաշվել 10մ բարձրությամբ և 4մ հիմքի տրամագծով գլանի մակերեսը:

Ըստ s = 2*pi*r*(h + r) բանաձևի

let s, d, r, h, pi;

v = 4;

r = d / 2;

h = 10;

pi = Math.PI;

console.log(s);

3. Հաշվել 3 և 4 կողմերով ուղղանկյուն եռանկյան անկյունագիծը: Կարող եք օգտվել Math.pow() ֆունկցիայից:

Ըստ c = a2 + b2 բանաձևի

let a, b, c;

a = 3;

b = 4;

c = Math.sqrt(Math.pow(a,2) + Math.pow(b,2));

console.log(c);
